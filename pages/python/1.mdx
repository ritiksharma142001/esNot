export const hue = (
    <div className="flex h-6 items-center gap-2">
        <input
            type="range"
            min="0"
            max="360"
            step="1"
            onChange={e => {
                const value = `${e.target.value}deg`
                e.target.nextSibling.textContent = value
                document.documentElement.style.setProperty(
                    '--nextra-primary-hue',
                    value
                )
            }}
        />
        <code className="w-14 text-sm text-gray-500" />
    </div>
)
